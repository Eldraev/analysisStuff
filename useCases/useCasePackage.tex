\usepackage{enumitem}

\usepackage[a4paper, top=1.25in, bottom=1.25in, left=2.25in, right=1.25in, marginparsep=0.25in, marginparwidth=1.25in]{geometry}

% For parts of the use case that is not a list.
% The argument is the headline in the margin.
%
% \begin{cpart}{Description}
%   Some description
% \end{cpart}
%
\newenvironment{cpart}[1] {
  \reversemarginpar
  \marginpar{
    \vspace{3.2mm}
    \textbf{#1}
  }
}
{\vspace{3mm}}

% For parts of the use case that is a list.
% The argument is the headline in the margin.
%
% \begin{cpartList}{Basic Flow}
%   \item Some flow
% \end{cpartList}
%
\newenvironment{cpartList}[1] {
  \reversemarginpar
  \marginpar{
    \vspace{3mm}
    \textbf{#1}
  }
  \begin{enumerate}[leftmargin=*]
}
{\end{enumerate}\vspace{3mm}}

% For the inner lists in the alternate flows and
% extensions. The arguments are number, letter, and
% the "headline item", e.g.
%
% \begin{innerList}{1}{a}{Alternate Flow}
%
\newenvironment{innerList}[3] {
  \item[#1 #2)] #3
  \begin{enumerate}[label=\arabic*., leftmargin=*]
}
{\end{enumerate}\vspace{3mm}}

% For the sequence diagram images. The argument is
% the image name, with or without the extension. For
% this one to work, the diagrams needs to be inside a
% sub directory called "sequenceDiagrams".
%
% \seq{login}
%
\newenvironment{seq}[1] {
\newgeometry{top=1in, left=1in, bottom=1in, right=1in}
\begin{center}
\includegraphics[scale=0.6]{sequenceDiagrams/#1}
\end{center}
\restoregeometry
\clearpage
}
{}
